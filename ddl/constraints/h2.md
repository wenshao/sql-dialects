> - ============================================================
> - PRIMARY KEY
> - ============================================================

```sql
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(64) NOT NULL
);
```

## 复合主键

```sql
CREATE TABLE order_items (
    order_id INT NOT NULL,
    item_id  INT NOT NULL,
    quantity INT,
    PRIMARY KEY (order_id, item_id)
);
```

## 命名主键

```sql
ALTER TABLE users ADD CONSTRAINT pk_users PRIMARY KEY (id);
```

## UNIQUE


```sql
CREATE TABLE members (
    id       INT PRIMARY KEY,
    email    VARCHAR(128) UNIQUE,
    phone    VARCHAR(20),
    CONSTRAINT uq_phone UNIQUE (phone)
);

ALTER TABLE members ADD CONSTRAINT uq_email UNIQUE (email);
ALTER TABLE members DROP CONSTRAINT uq_email;
```

## NOT NULL


```sql
CREATE TABLE products (
    id    INT PRIMARY KEY,
    name  VARCHAR(100) NOT NULL,
    price DECIMAL(10,2) NOT NULL
);

ALTER TABLE products ALTER COLUMN name SET NOT NULL;
ALTER TABLE products ALTER COLUMN name DROP NOT NULL;
```

## DEFAULT


```sql
CREATE TABLE events (
    id         IDENTITY,
    status     INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE events ALTER COLUMN status SET DEFAULT 1;
ALTER TABLE events ALTER COLUMN status DROP DEFAULT;
```

## CHECK


```sql
CREATE TABLE employees (
    id     INT PRIMARY KEY,
    age    INT CHECK (age BETWEEN 16 AND 150),
    salary DECIMAL(10,2),
    CONSTRAINT chk_salary CHECK (salary >= 0)
);

ALTER TABLE employees ADD CONSTRAINT chk_age CHECK (age >= 16);
ALTER TABLE employees DROP CONSTRAINT chk_age;
```

## FOREIGN KEY


```sql
CREATE TABLE orders (
    id      INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    amount  DECIMAL(10,2),
    CONSTRAINT fk_user FOREIGN KEY (user_id) REFERENCES users(id)
);
```

## 级联操作

```sql
CREATE TABLE comments (
    id      INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    content VARCHAR(1000),
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
```

## ON DELETE 选项: CASCADE, SET NULL, SET DEFAULT, RESTRICT, NO ACTION

```sql
ALTER TABLE orders ADD CONSTRAINT fk_user
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE;
ALTER TABLE orders DROP CONSTRAINT fk_user;
```

## 查看约束


```sql
SELECT * FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE TABLE_NAME = 'USERS';
SELECT * FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE WHERE TABLE_NAME = 'USERS';
SELECT * FROM INFORMATION_SCHEMA.CHECK_CONSTRAINTS;
SELECT * FROM INFORMATION_SCHEMA.REFERENTIAL_CONSTRAINTS;
```

注意：H2 支持所有标准 SQL 约束
注意：约束都是强制执行的
注意：支持 ON DELETE / ON UPDATE 级联操作
注意：支持命名约束和匿名约束
