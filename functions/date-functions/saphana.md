# SAP HANA: Date Functions

## Current date/time

```sql
SELECT CURRENT_DATE FROM DUMMY;
SELECT CURRENT_TIME FROM DUMMY;
SELECT CURRENT_TIMESTAMP FROM DUMMY;
SELECT NOW() FROM DUMMY;                                -- same as CURRENT_TIMESTAMP
SELECT CURRENT_UTCDATE FROM DUMMY;                      -- UTC date
SELECT CURRENT_UTCTIMESTAMP FROM DUMMY;                  -- UTC timestamp
```

## Date construction

```sql
SELECT TO_DATE('2024-01-15', 'YYYY-MM-DD') FROM DUMMY;
SELECT TO_TIMESTAMP('2024-01-15 10:30:00', 'YYYY-MM-DD HH24:MI:SS') FROM DUMMY;
SELECT TO_TIME('10:30:00', 'HH24:MI:SS') FROM DUMMY;
SELECT TO_SECONDDATE('2024-01-15 10:30:00', 'YYYY-MM-DD HH24:MI:SS') FROM DUMMY;
```

## Date arithmetic (dedicated functions)

```sql
SELECT ADD_DAYS(CURRENT_DATE, 7) FROM DUMMY;
SELECT ADD_MONTHS(CURRENT_DATE, 3) FROM DUMMY;
SELECT ADD_YEARS(CURRENT_DATE, 1) FROM DUMMY;
SELECT ADD_HOURS(CURRENT_TIMESTAMP, 2) FROM DUMMY;
SELECT ADD_MINUTES(CURRENT_TIMESTAMP, 30) FROM DUMMY;
SELECT ADD_SECONDS(CURRENT_TIMESTAMP, 3600) FROM DUMMY;
SELECT ADD_WORKDAYS(CURRENT_DATE, 5, 'US') FROM DUMMY;  -- skip weekends/holidays
```

## Date difference

```sql
SELECT DAYS_BETWEEN('2024-01-01', '2024-12-31') FROM DUMMY;         -- 365
SELECT MONTHS_BETWEEN('2024-01-01', '2024-06-01') FROM DUMMY;       -- 5
SELECT YEARS_BETWEEN('2020-01-01', '2024-01-01') FROM DUMMY;        -- 4
SELECT SECONDS_BETWEEN(CURRENT_TIMESTAMP, ADD_HOURS(CURRENT_TIMESTAMP, 1)) FROM DUMMY;
SELECT NANO100_BETWEEN(CURRENT_TIMESTAMP, ADD_SECONDS(CURRENT_TIMESTAMP, 1)) FROM DUMMY;
SELECT WORKDAYS_BETWEEN('2024-01-01', '2024-01-31', 'US') FROM DUMMY;
```

## Extract

```sql
SELECT EXTRACT(YEAR FROM CURRENT_DATE) FROM DUMMY;
SELECT EXTRACT(MONTH FROM CURRENT_DATE) FROM DUMMY;
SELECT EXTRACT(DAY FROM CURRENT_DATE) FROM DUMMY;
SELECT EXTRACT(HOUR FROM CURRENT_TIMESTAMP) FROM DUMMY;
```

## Convenience extraction functions

```sql
SELECT YEAR(CURRENT_DATE) FROM DUMMY;
SELECT MONTH(CURRENT_DATE) FROM DUMMY;
SELECT DAYOFMONTH(CURRENT_DATE) FROM DUMMY;
SELECT HOUR(CURRENT_TIMESTAMP) FROM DUMMY;
SELECT MINUTE(CURRENT_TIMESTAMP) FROM DUMMY;
SELECT SECOND(CURRENT_TIMESTAMP) FROM DUMMY;
SELECT DAYOFWEEK(CURRENT_DATE) FROM DUMMY;          -- 0=Monday
SELECT DAYOFYEAR(CURRENT_DATE) FROM DUMMY;
SELECT WEEK(CURRENT_DATE) FROM DUMMY;
SELECT ISOWEEK_NUMBER(CURRENT_DATE) FROM DUMMY;
SELECT QUARTER(CURRENT_DATE) FROM DUMMY;
SELECT DAYNAME(CURRENT_DATE) FROM DUMMY;             -- 'MONDAY'
SELECT MONTHNAME(CURRENT_DATE) FROM DUMMY;           -- 'JANUARY'
```

## Formatting

```sql
SELECT TO_VARCHAR(CURRENT_TIMESTAMP, 'YYYY-MM-DD HH24:MI:SS') FROM DUMMY;
SELECT TO_NVARCHAR(CURRENT_DATE, 'YYYY/MM/DD') FROM DUMMY;
```

## Last day of month

```sql
SELECT LAST_DAY(CURRENT_DATE) FROM DUMMY;
```

## Next day (specific weekday)

```sql
SELECT NEXT_DAY(CURRENT_DATE) FROM DUMMY;            -- next day
```

## UTCTOLOCAL / LOCALTOUTC (timezone conversion)

```sql
SELECT UTCTOLOCAL(CURRENT_UTCTIMESTAMP, 'US/Eastern') FROM DUMMY;
SELECT LOCALTOUTC(CURRENT_TIMESTAMP, 'US/Eastern') FROM DUMMY;
```

## Series functions (time series)

```sql
SELECT SERIES_GENERATE_DATE('INTERVAL 1 DAY', '2024-01-01', '2024-01-31') FROM DUMMY;
SELECT SERIES_GENERATE_TIMESTAMP('INTERVAL 1 HOUR', '2024-01-01', '2024-01-02') FROM DUMMY;
```

Note: ADD_WORKDAYS / WORKDAYS_BETWEEN consider business calendars
Note: DUMMY is SAP HANA's single-row table
Note: SECONDDATE is unique to SAP HANA (compact second-precision)
Note: SERIES_GENERATE functions create date/time sequences
