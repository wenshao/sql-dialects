# Vertica: 日期函数

> 参考资料:
> - [Vertica SQL Reference](https://www.vertica.com/docs/latest/HTML/Content/Authoring/SQLReferenceManual/SQLReferenceManual.htm)
> - [Vertica Functions](https://www.vertica.com/docs/latest/HTML/Content/Authoring/SQLReferenceManual/Functions/Functions.htm)


当前日期时间
```sql
SELECT NOW();                                -- TIMESTAMPTZ
SELECT CURRENT_TIMESTAMP;                    -- TIMESTAMPTZ
SELECT CURRENT_DATE;                         -- DATE
SELECT CURRENT_TIME;                         -- TIMETZ
SELECT GETDATE();                            -- Vertica 扩展
SELECT SYSDATE();                            -- 实际执行时间
SELECT CLOCK_TIMESTAMP();                    -- 实际执行时间
```


构造日期
```sql
SELECT DATE '2024-01-15';
SELECT TIMESTAMP '2024-01-15 10:30:00';
SELECT TO_TIMESTAMP('2024-01-15', 'YYYY-MM-DD');
SELECT TO_DATE('2024-01-15', 'YYYY-MM-DD');
```


日期加减
```sql
SELECT TIMESTAMPADD(DAY, 7, DATE '2024-01-15');
SELECT TIMESTAMPADD(MONTH, 3, CURRENT_DATE);
SELECT TIMESTAMPADD(HOUR, 2, CURRENT_TIMESTAMP);
SELECT ADD_MONTHS(DATE '2024-01-15', 3);
SELECT DATE '2024-01-15' + INTERVAL '7 days';
SELECT DATE '2024-01-15' + 7;               -- 加 7 天
SELECT NOW() - INTERVAL '1 month';
```


日期差
```sql
SELECT DATEDIFF('day', DATE '2024-01-01', DATE '2024-12-31');     -- 365
SELECT DATEDIFF('month', DATE '2024-01-01', DATE '2024-12-31');   -- 11
SELECT DATEDIFF('hour', TIMESTAMP '2024-01-01', TIMESTAMP '2024-01-02 12:00:00');
SELECT TIMESTAMPDIFF(MONTH, DATE '2024-01-01', DATE '2024-06-15');
SELECT AGE_IN_MONTHS(DATE '2024-12-31', DATE '2024-01-01');
SELECT AGE_IN_YEARS(DATE '2024-12-31', DATE '2020-01-01');
SELECT MONTHS_BETWEEN(DATE '2024-12-31', DATE '2024-01-01');
```


提取
```sql
SELECT EXTRACT(YEAR FROM TIMESTAMP '2024-01-15 10:30:00');
SELECT EXTRACT(MONTH FROM CURRENT_DATE);
SELECT EXTRACT(DAY FROM CURRENT_DATE);
SELECT EXTRACT(HOUR FROM CURRENT_TIMESTAMP);
SELECT EXTRACT(DOW FROM DATE '2024-01-15');   -- 0=周日（同 PostgreSQL）
SELECT EXTRACT(DOY FROM DATE '2024-01-15');
SELECT EXTRACT(WEEK FROM DATE '2024-01-15');
SELECT EXTRACT(QUARTER FROM DATE '2024-06-15');
SELECT EXTRACT(EPOCH FROM NOW());
SELECT DATE_PART('year', NOW());
SELECT YEAR(DATE '2024-01-15');
SELECT MONTH(DATE '2024-01-15');
SELECT DAY(DATE '2024-01-15');
SELECT HOUR(CURRENT_TIMESTAMP);
SELECT DAYOFWEEK(DATE '2024-01-15');
SELECT DAYOFYEAR(DATE '2024-01-15');
SELECT QUARTER(DATE '2024-06-15');
SELECT WEEK(DATE '2024-01-15');
```


格式化
```sql
SELECT TO_CHAR(NOW(), 'YYYY-MM-DD HH24:MI:SS');
SELECT TO_CHAR(NOW(), 'Day, DD Month YYYY');
```


截断
```sql
SELECT DATE_TRUNC('year', NOW());
SELECT DATE_TRUNC('quarter', NOW());
SELECT DATE_TRUNC('month', NOW());
SELECT DATE_TRUNC('week', NOW());
SELECT DATE_TRUNC('day', NOW());
SELECT DATE_TRUNC('hour', NOW());
SELECT TRUNC(NOW()::DATE);
```


时区
```sql
SELECT NOW() AT TIME ZONE 'UTC';
SELECT NOW() AT TIME ZONE 'Asia/Shanghai';
```


月末 / 月初
```sql
SELECT LAST_DAY(DATE '2024-02-15');          -- 2024-02-29
SELECT NEXT_DAY(DATE '2024-01-15', 'Monday');
```


注意：Vertica 同时支持 PostgreSQL 和 Oracle 风格的日期函数
注意：DATEDIFF 参数顺序与 MySQL 不同（单位在前）
注意：AGE_IN_MONTHS / AGE_IN_YEARS 是 Vertica 特有函数
注意：EXTRACT DOW 从 0（周日）开始（同 PostgreSQL）
